function canAssign(dj, dateStr, slot, targetMap) {
  if (dj !== 'Guest DJ' && submittedDJs.get(dj) !== 'submitted') return false;
  const unavailDate = getUnavailLookupDate(dateStr, slot);
  if (unavailabilityMap[dj]?.includes(`${unavailDate}|${slot}`)) return false;
  if (targetMap[`${dateStr}|${slot}`]) return false;
  return true;
}
